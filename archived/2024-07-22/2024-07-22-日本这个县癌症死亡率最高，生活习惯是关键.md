width: 54.6177%; height: 597px; border-width: 0px; margin-left: auto; margin-right: auto;"><colgroup><col style="width: 100%;"></colgroup><tbody><tr style="height: 566.766px; text-align: left;"><td style="height: 566.766px;"><img src="https://cn.nikkei.com/images/2024/07/0717/0717-12-2-LL.jpg" loading="lazy" width="600" height="557" data-path="local-images:/2024/07/0717/0717-12-2-LL.jpg" referrerpolicy="no-referrer"></td></tr><tr style="height: 30.2344px;"><td style="height: 30.2344px;
